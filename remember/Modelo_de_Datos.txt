Campo expense: id(number), trip_id(number), concept(string), amount(number), category(string), date(date), created_at(date).
Campo CheckListItem: id(number), trip_id(number), item_name(string), priority(number), completed(boolean), created_at(date).
Campo TripMember: user_id(number), trip_id(number), created_at(date).
